üíª How to Listen for Events 
// Listen for component ready event
document.addEventListener('wb-xtest:ready', (e) => {
  console.log('Component ready:', e.target);
  console.log('Variant:', e.target.getAttribute('variant'));
});

// Listen for variant change event
document.addEventListener('wb-xtest:change', (e) => {
  console.log('Variant changed:', e.detail);
});

// Listen for clicks on wb-xtest components
document.addEventListener('click', (e) => {
  const component = e.target.closest('wb-xtest');
  if (component) {
    console.log('Clicked:', component.textContent);
  }
});
