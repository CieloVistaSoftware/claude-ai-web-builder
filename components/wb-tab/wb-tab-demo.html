const tabComponent = document.getElementById('dynamic-tabs');

// Add new tab
tabComponent.addTab({
    id: 'new-tab',
    label: 'New Tab',
    content: '<p>Dynamic content</p>',
    icon: '‚ú®',
    closable: true
});

// Switch to tab
tabComponent.setActiveTab('profile');

// Listen for changes
tabComponent.addEventListener('tab-change', (e) => {
    console.log('Tab changed to:', e.detail.tabId);
});
