JavaScript API

// Get tab component
const tabComponent = document.getElementById('my-tabs');

// Add new tab dynamically
tabComponent.addTab({
  id: 'new-tab',
  label: 'New Tab',
  content: '<p>Dynamic content</p>',
  icon: '‚ú®',
  closable: true
});

// Switch to specific tab
tabComponent.setActiveTab('profile');

// Listen for tab changes
tabComponent.addEventListener('tab-change', (e) => {
  console.log('Tab changed to:', e.detail.tabId);
});

// Listen for tab close events
tabComponent.addEventListener('tab-close', (e) => {
  console.log('Tab closed:', e.detail.tabId);
});
