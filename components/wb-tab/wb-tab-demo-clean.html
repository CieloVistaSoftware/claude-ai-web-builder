Configuration

Attributes:
active-tab    - Index or ID of initially active tab
theme         - light | dark | auto (default: auto)
orientation   - horizontal | vertical (default: horizontal)
variant       - default | pills | underline | card
closable      - Allow tabs to be closed
lazy-load     - Only render content when tab becomes active
scrollable    - Enable horizontal scrolling for overflow
animated      - Enable smooth transitions
disabled      - Disable the entire component

JavaScript API

// Get tab element
const tabComponent = document.querySelector('wb-tab');

// Switch to a specific tab
tabComponent.switchTo('tab-id');

// Listen for tab changes
tabComponent.addEventListener('wb-tab-change', (e) => {
  console.log('Active tab:', e.detail.activeTab);
  console.log('Previous tab:', e.detail.previousTab);
});
