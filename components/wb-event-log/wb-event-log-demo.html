Dispatch Events

// Info event
document.dispatchEvent(new CustomEvent('wb:info', {
  detail: { message: 'Hello World' }
}));

// Warning event
document.dispatchEvent(new CustomEvent('wb:warning', {
  detail: { message: 'Something needs attention' }
}));

// Error event
document.dispatchEvent(new CustomEvent('wb:error', {
  detail: { message: 'Something went wrong' }
}));

// Success event
document.dispatchEvent(new CustomEvent('wb:success', {
  detail: { message: 'Operation completed' }
}));

Available Events

  • wb:info - Information messages (blue)
  • wb:warning - Warning messages (orange)
  • wb:error - Error messages (red)
  • wb:success - Success messages (green)
  • wb:debug - Debug messages (gray)
  • wb:user - User action messages (purple)
