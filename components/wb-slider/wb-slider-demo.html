Usage Examples

HTML Usage

<!-- Basic slider -->
<wb-slider id="volume" label="Volume" min="0" max="100" value="50"></wb-slider>

<!-- With unit and custom range -->
<wb-slider id="spacing" label="Spacing" min="0" max="50" 
           value="16" unit="px" show-ticks="true"></wb-slider>

<!-- Vertical orientation -->
<wb-slider orientation="vertical" label="Height" 
           min="0" max="200" value="100"></wb-slider>

JavaScript Usage

// Create slider programmatically
const slider = WBSlider.create({
    label: 'Opacity',
    min: 0,
    max: 1,
    step: 0.1,
    value: 0.8,
    unit: '',
    onchange: (value) => console.log('New value:', value)
});

// API methods
WBSlider.setValue(slider, 0.5);
const currentValue = WBSlider.getValue(slider);
WBSlider.setDisabled(slider, true);
WBSlider.setRange(slider, 0, 2);
